def initialize(details = {})

    self.public_send("#{attr}=", value)
  end if details
  @details = details

def to_h
  hash = {}
  PRODUCT_KEYS.each do |key|
    value = self.public_send(key)
    hash[key.to_s.classify] = value unless value.nil?
  end
  hash
end
